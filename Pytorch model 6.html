# Train the network

train_losses = []
val_losses = []
num_epochs = 300
batch_size = 64
num_batches = len(X_train_norm) // batch_size

for epoch in range(num_epochs):
    # Shuffle the data
    permutation = torch.randperm(X_train_norm.shape[0])
    X_train_norm_shuffled = X_train_norm[permutation]
    y_train_shuffled = y_train[permutation]
    
    # Iterate over batches
    for i in range(num_batches):
        # Get batch
        start = i * batch_size
        end = (i+1) * batch_size
        batch_X = X_train_norm_shuffled[start:end]
        batch_y = y_train_shuffled[start:end]
        
        # Forward pass and calculate loss
        outputs = model(batch_X)
        loss = criterion(outputs, batch_y)
        
        # Backward and optimize
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        train_acc = ((outputs > 0.5) == batch_y).float().mean().item()
    
    # Evaluate on validation set
    with torch.no_grad():
        val_outputs = model(X_val_norm)
        val_loss = criterion(val_outputs, y_val)
#         lr_scheduler.step(val_loss)
        val_acc = ((val_outputs > 0.5) == y_val).float().mean().item()
        
    # Print progress
    if (epoch+1) % 10 == 0:
        train_losses.append(loss.item())
        val_losses.append(val_loss.item())
        print(f'Epoch [{epoch+1}/{num_epochs}] --> Train Loss: {loss.item():.4f}, Train acc: {train_acc}, Val Loss: {val_loss.item():.4f}, Val Acc: {val_acc:.4f}')
